#### Usage

```css title="global.css"
@property --border-angle {
  inherits: false;
  initial-value: 0deg;
  syntax: '<angle>';
}
```

```ts title="tailwind.config.ts" {5,8-10}
tailwind.config = {
  theme: {
    extend: {
      animation: {
        border: 'border 4s linear infinite',
      },
      keyframes: {
        border: {
          to: { '--border-angle': '360deg' },
        },
      },
    },
  },
}
```
